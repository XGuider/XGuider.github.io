import java.util.*;

class Solution {
    // 使用成员变量，方便在方法间共享
    private Set<String> wordSet;
    private Map<Integer, List<String>> memo;

    public List<String> wordBreak(String s, String[] wordDict) {
        this.wordSet = new HashSet<>();
        for (String word : wordDict) {
            this.wordSet.add(word); // 修正1: 添加 word 而不是 wordDict[i]
        }
        this.memo = new HashMap<>(); // 初始化记忆化缓存

        // 开始回溯搜索，从索引 0 开始
        return backtrack(s, 0);
    }

    /**
     * 递归函数，返回从 startIndex 开始到字符串末尾的所有可能拆分结果
     */
    private List<String> backtrack(String s, int startIndex) {
        // 1. 检查备忘录，如果已经计算过，直接返回结果
        if (memo.containsKey(startIndex)) {
            return memo.get(startIndex);
        }

        // 2. 初始化当前子问题的结果列表
        List<String> currentResult = new ArrayList<>();

        // 3. 如果到达字符串末尾，说明找到了一种有效的拆分方式（一个空的句子片段）
        if (startIndex == s.length()) {
            currentResult.add(""); // 添加一个空字符串作为递归的基石
            return currentResult;
        }

        // 4. 遍历所有可能的结束位置
        for (int endIndex = startIndex + 1; endIndex <= s.length(); endIndex++) {
            String prefix = s.substring(startIndex, endIndex);

            // 5. 如果当前前缀在字典中
            if (wordSet.contains(prefix)) {
                // 6. 递归求解剩余部分的拆分结果
                List<String> suffixes = backtrack(s, endIndex);

                // 7. 将当前前缀与所有可能的后缀组合起来
                for (String suffix : suffixes) {
                    // 如果后缀为空（说明已经到达末尾），则不需要加空格
                    if (suffix.isEmpty()) {
                        currentResult.add(prefix);
                    } else {
                        currentResult.add(prefix + " " + suffix);
                    }
                }
            }
        }

        // 8. 将当前子问题的结果存入备忘录
        memo.put(startIndex, currentResult);
        return currentResult;
    }
}
