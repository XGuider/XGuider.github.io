给你一个整数数组 nums，数组中的元素 互不相同。返回该数组所有可能的 子集（幂集）。解集 不能 包含重复的子集。你可以按 任意顺序 返回解集。
子集（Subset）是指从原集合中选取部分（可以是全部，也可以是空集）元素组成的集合。例如，集合 {1, 2} 的子集有 {}, {1}, {2}, {1, 2}。

示例 1：
输入: nums = [1,2,3]
输出: [[],[1],[2],[1,2],[3],[1,3],[2,3],[1,2,3]]


class Solution {
	public List<List<Integer>> subsets (int[] nums) {
		List<List<Integer>> result = new ArrayList<>();
		if (nums.length = =0 || nums == null) {return result;}
		List<Integer> tmp = new ArrayList<>();
		dfs(nums, result, tmp);
		return result;
	}

	public void dfs(int[] nums, List<List<Integer>> result, List<Integer> tmp) {

		if (result.contains(tmp) || tmp.size() > nums.length) {return;}
		result.add(new ArrayList<>(tmp));
		for (int i = 0; i < nums.length; i++ ) {
			if (tmp.contains(nums[i])) continue; // 没有必要；
			tmp.add(nums[i]);
			dfs(nums, result, tmp);
			tmp.remove(tmp.size() - 1);
		}
	} 
}



# 标准答案

import java.util.ArrayList;
import java.util.List;

class Solution {
    public List<List<Integer>> subsets(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        if (nums == null) { // 先检查 null
            return result;
        }
        List<Integer> path = new ArrayList<>(); // 用于存储当前正在构建的子集
        // 从索引 0 开始进行回溯
        backtrack(nums, 0, path, result);
        return result;
    }

    /**
     * 回溯函数
     * @param nums 原始数组
     * @param start 当前考虑的元素的起始索引
     * @param path 当前正在构建的子集
     * @param result 最终的结果集合
     */
    private void backtrack(int[] nums, int start, List<Integer> path, List<List<Integer>> result) {
        
        // 将当前路径的副本加入结果集
        // 这是收集子集的关键步骤，每次进入函数时都添加一次
        result.add(new ArrayList<>(path));

        // 从 start 开始遍历，确保不回头，避免重复
        for (int i = start; i < nums.length; i++) {
            
            // 做选择：将 nums[i] 加入当前路径
            path.add(nums[i]);

            // 递归：基于新的路径，从下一个位置开始继续构建
            backtrack(nums, i + 1, path, result);

            // 撤销选择：将刚才加入的 nums[i] 移出路径，进行回溯
            path.remove(path.size() - 1);
        }
    }
}
