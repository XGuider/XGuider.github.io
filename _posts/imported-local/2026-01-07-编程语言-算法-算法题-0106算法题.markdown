给定一个字符串 s，请你找出其中不含有重复字符的 最长子串 的长度。
输入: s = "abcabcbb"

输出: 3

解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3


import java.util.List;
class Solution {
	public int lengthOfLongestSubstring(String s) {
		int result = 0;
		if (s == null) {return 0;}
		if (s.length() == 1) {return 1;}
		for(int i = 0; i < s.length()-1; i++) {
			List<Character>() al = new ArrayList<>(); -- al.contains() 操作: O(n) （因为 ArrayList 查找是线性查找）
			al.add(s.charAt(i))
			for(int j = i+1; j < s.length(); j++) {
				if (al.contains(s.charAt(j))) {
					break;
				}
				al.add(s.charAt(j));
			}
			result = al.size() > result? al.size(): result
		}
		return result;
	}
}

import java.util.HashSet;
import java.util.Set;

class Solution {
    public int lengthOfLongestSubstring(String s) {
        int result = 0;
        if (s == null || s.length() == 0) { // 修复空字符串处理
            return 0;
        }

        for (int i = 0; i < s.length(); i++) {
            Set<Character> seen = new HashSet<>();
            for (int j = i; j < s.length(); j++) {
                if (seen.contains(s.charAt(j))) {
                    break; // 遇到重复字符，跳出内层循环
                }
                seen.add(s.charAt(j));
            }
            // 内层循环结束后，更新结果
            result = Math.max(result, seen.size());
        }
        return result;
    }
}
推荐解法：滑动窗口（O(n)，最优解）
import java.util.HashSet;
import java.util.Set;

class Solution {
    public int lengthOfLongestSubstring(String s) {
        if (s == null || s.length() == 0) {
            return 0;
        }

        Set<Character> window = new HashSet<>();
        int left = 0;
        int right = 0;
        int result = 0;

        while (right < s.length()) {
            // 如果当前字符已存在于窗口中，则收缩左边界
            while (window.contains(s.charAt(right))) {
                window.remove(s.charAt(left));
                left++;
            }
            // 将当前字符加入窗口
            window.add(s.charAt(right));
            // 更新最大长度
            result = Math.max(result, right - left + 1);
            // 扩展右边界
            right++;
        }

        return result;
    }
}
