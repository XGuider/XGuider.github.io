今日算法题（Day 30）
题目编号： LeetCode 188
题目名称： 买卖股票的最佳时机 IV
难度： 🔴 困难
标签： 数组、动态规划
📝 题目描述
给你一个整数数组 prices 和一个整数 k，其中 prices[i] 是某支给定的股票在第 i 天的价格。
设计一个算法来计算你所能获取的最大利润。你最多可以完成 k 笔交易。也就是说，你最多可以买 k 次，卖 k 次。
注意：你不能同时参与多笔交易（你必须在再次购买前出售掉之前的股票）。


class Solution {
    public int maxProfit(int k, int[] prices) {
        if (prices == null || prices.length <= 1 || k == 0) {
            return 0;
        }

        int n = prices.length;

        // 如果 k 很大，超过了理论最大交易次数 n/2，那么问题就退化为无限次交易
        // 这样可以避免不必要的空间浪费
        if (k > n / 2) {
            // 可以在这里调用昨天的贪心解法
            int totalProfit = 0;
            for (int i = 1; i < n; i++) {
                if (prices[i] > prices[i - 1]) {
                    totalProfit += prices[i] - prices[i - 1];
                }
            }
            return totalProfit;
        }

        // dp 数组，大小为 2*k+1
        // dp[2*j] 表示完成 j 次交易且不持股的最大利润
        // dp[2*j+1] 表示完成 j 次交易且持股的最大利润
        int[] dp = new int[2 * k + 1];

        // 初始化：第0天
        // 所有不持股状态（偶数索引）保持为 0
        // 所有持股状态（奇数索引）初始化为 -prices[0]
        for (int j = 1; j <= 2 * k; j += 2) {
            dp[j] = -prices[0];
        }

        // 从第1天开始遍历
        for (int i = 1; i < n; i++) {
            // 遍历所有可能的交易次数 j (从 1 到 k)
            for (int j = 1; j <= k; j++) {
                // 第 j 次买入后的状态 (对应 dp 数组的奇数索引 2*j-1)
                dp[2 * j - 1] = Math.max(dp[2 * j - 1], dp[2 * j - 2] - prices[i]);
                
                // 第 j 次卖出后的状态 (对应 dp 数组的偶数索引 2*j)
                dp[2 * j] = Math.max(dp[2 * j], dp[2 * j - 1] + prices[i]);
            }
        }

        // 返回完成 k 次交易且不持股的最大利润
        return dp[2 * k];
    }
}
