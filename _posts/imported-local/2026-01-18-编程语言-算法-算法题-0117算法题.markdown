题目名称： 爬楼梯
难度： 🟢 简单
标签： 记忆化搜索、数学、动态规划


class Solution {. # 纯递归（暴力解法）
	public Integer climbStairs(int n){
		if (n == 1) return 1;
		if (n == 2) return 2;
		return climbStairs(n-1) + climbStairs(n-2);

	}
}

public class Solution {
    public int climbStairs(int n) {
        // 处理边界情况
        if (n <= 2) {
            return n;
        }

        // 使用两个变量来存储前两个状态，节省空间
        // prev2 代表 dp[i-2]
        int prev2 = 1; 
        // prev1 代表 dp[i-1] 
        int prev1 = 2; 
        int current = 0;

        // 从第 3 阶开始迭代计算
        for (int i = 3; i <= n; i++) {
            // dp[i] = dp[i-1] + dp[i-2]
            current = prev1 + prev2; 
            // 更新状态，为下一次迭代准备
            prev2 = prev1; 
            prev1 = current;
        }

        return current;
    }
}
