# 登录 MySQL
mysql -u root -p
-- 创建用户
CREATE USER 'ai'@'localhost' IDENTIFIED BY 'ai';

-- 授予 data_agent 数据库的所有权限
GRANT ALL PRIVILEGES ON data_agent.* TO 'ai'@'localhost';

-- 刷新权限
FLUSH PRIVILEGES;


# 查看用户权限
SELECT 
    User, 
    Host, 
    Db AS DatabaseName,
    GROUP_CONCAT(
        CASE WHEN Select_priv = 'Y' THEN 'SELECT' END,
        CASE WHEN Insert_priv = 'Y' THEN 'INSERT' END,
        CASE WHEN Update_priv = 'Y' THEN 'UPDATE' END,
        CASE WHEN Delete_priv = 'Y' THEN 'DELETE' END,
        CASE WHEN Create_priv = 'Y' THEN 'CREATE' END,
        CASE WHEN Drop_priv = 'Y' THEN 'DROP' END,
        CASE WHEN Grant_priv = 'Y' THEN 'GRANT' END
    ) AS Privileges
FROM mysql.db
GROUP BY User, Host, Db;
