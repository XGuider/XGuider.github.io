n 皇后问题研究的是如何将 n 个皇后放置在 n×n 的棋盘上，并且使皇后彼此之间不能相互攻击。
给你一个整数 n，返回所有不同的 n 皇后问题的解决方案。
每一种解法包含一个不同的 n 皇后问题的棋子放置方案，该方案中 'Q' 和 '.' 分别代表了皇后和空位。
💡 示例
示例 1：
输入： n = 4
输出： [[".Q..","...Q","Q...","..Q."],["..Q.","Q...","...Q",".Q.."]]
解释： 如上图所示，4 皇后问题存在两个不同的解法。
示例 2：
输入： n = 1
输出： [["Q"]]






class Solution {
	public List<String> solveNQueens (int n){

		List<String> result = new ArrayList<>();
		char[][] qipan = new char[n][n];
		for (int i = 0; i < n; i++ ) {
			for (int j = 0; j < n; j++ ) {
				qipan[i][j] = '.';
			}
		}
		dfs(qipan, result , 0);
		return result;
		}


		public boolean dfs(char[][] qipan, List<String> result, int n){
			for (int i = n; i < qipan.length; i++ ) {
				for (int j = 0; j < qipan.length; j++ ) {

					if (solve(qipan, 'Q', i ,j)) {
						return false;
					}

					qipan[i][j] = 'Q';
					if (dfs(qipan, result, n + 1 )) {
						for (int i = 0; i < n; i++ ) {
							result.add(qipan[i].stream().reduce((a, b) -> a + b));
						}
					}
					qipan[i][j] = '.';
				}
			}
			return false;
	}

	public boolean solve (char[][] qipan, char ch, int i, int j) {
		for (int j = 0 ; j < qipan.length; j++) {
			if (qipan[i][j] == ch) {
				return false;
			}
		}

		for (int i = 0 ; i < qipan.length; i++) {
			if (qipan[i][j] == ch) {
				return false;
			}
		}

import java.util.*;

class Solution {
    public List<List<String>> solveNQueens(int n) {
        List<List<String>> result = new ArrayList<>();
        char[][] board = new char[n][n];
        for (int i = 0; i < n; i++) {
            Arrays.fill(board[i], '.'); // 填充‘.’
        }
        backtrack(result, board, 0, n); // 从第 0 行开始
        return result;
    }

    private void backtrack(List<List<String>> result, char[][] board, int row, int n) {
        if (row == n) { // 所有行都已放置成功
            List<String> solution = new ArrayList<>();
            for (char[] r : board) {
                solution.add(new String(r)); // char[] 转String
            }
            result.add(solution);
            return;
        }

        for (int col = 0; col < n; col++) { // 遍历当前行的每一列
            if (isValid(board, row, col, n)) { // 检查 (row, col) 是否可以放置
                board[row][col] = 'Q'; // 放置皇后
                backtrack(result, board, row + 1, n); // 递归处理下一行
                board[row][col] = '.'; // 撤销选择
            }
        }
    }

    private boolean isValid(char[][] board, int row, int col, int n) {
        // 检查列
        for (int i = 0; i < row; i++) {
            if (board[i][col] == 'Q') return false;
        }
        // 检查左上对角线
        for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
            if (board[i][j] == 'Q') return false;
        }
        // 检查右上对角线
        for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
            if (board[i][j] == 'Q') return false;
        }
        return true;
    }
}
