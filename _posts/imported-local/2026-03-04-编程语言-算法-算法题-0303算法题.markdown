题目编号： LeetCode 329
题目名称： 矩阵中的最长递增路径
难度： 🔴 困难
标签： 深度优先搜索、拓扑排序、记忆化搜索、动态规划

题目描述
给定一个 m x n 整数矩阵 matrix，找出其中 最长递增路径 的长度。
对于每个单元格，你可以往上、下、左、右四个方向移动。你 不能 在 对角线 方向上移动或移动到 边界外（即不允许环绕）。
💡 示例
示例 1：
输入： matrix = [[9,9,4],[6,6,8],[2,1,1]]
输出： 4
解释： 最长递增路径为 [1, 2, 6, 9]。

    0,1 
1,0 1,1 2,1
    1,2


class Solution {
    // 方向数组，用于上下左右移动
    private static final int[][] DIRECTIONS = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    private int[][] memo;
    private int m, n;

    public int longestIncreasingPath(int[][] matrix) {
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) {
            return 0;
        }

        this.m = matrix.length;
        this.n = matrix[0].length;
        this.memo = new int[m][n]; // 记忆化数组
        int maxPath = 0;

        // 遍历矩阵中的每一个点，作为起点进行DFS
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                // 更新全局最大路径长度
                maxPath = Math.max(maxPath, dfs(matrix, i, j));
            }
        }

        return maxPath;
    }

    /**
     * DFS函数：计算从 (row, col) 出发的最长递增路径长度
     */
    private int dfs(int[][] matrix, int row, int col) {
        // 1. 检查备忘录，如果已经计算过，直接返回
        if (memo[row][col] != 0) {
            return memo[row][col];
        }

        // 2. 初始化当前点的最长路径长度为 1（它本身）
        memo[row][col] = 1;

        // 3. 探索四个方向的邻居
        for (int[] dir : DIRECTIONS) {
            int newRow = row + dir[0];
            int newCol = col + dir[1];

            // 4. 检查边界和递增条件
            if (newRow >= 0 && newRow < m && newCol >= 0 && newCol < n &&
                matrix[newRow][newCol] > matrix[row][col]) {
                
                // 5. 递归计算邻居的最长路径，并更新当前点的最长路径
                memo[row][col] = Math.max(memo[row][col], dfs(matrix, newRow, newCol) + 1);
            }
        }

        // 6. 返回从 (row, col) 出发的最长路径长度
        return memo[row][col];
    }
}
