题目编号： LeetCode 120
题目名称： 三角形最小路径和
难度： 🟠 中等
标签： 数组、动态规划

给定一个三角形 triangle，找出自顶向下的最小路径和。
每一步只能移动到下一行中相邻的结点上。相邻的结点指的是下标与上一层结点下标相同或者等于上一层结点下标 + 1 的两个结点。
也就是说，如果正位于当前行的下标 i，那么下一步可以移动到下一行的下标 i 或 i + 1。
💡 示例
示例 1：
输入： triangle = [[2],[3,4],[6,5,7],[4,1,8,3]]
输出： 11
解释： 如下面简图所示：

import java.util.*;

class Solution {
    public int minimumTotal(int[][] triangle) {
        // 注意：此 DFS 方法会导致超时，仅为展示如何修正语法错误
        int row = triangle.length;
        if (triangle == null || row == 0) { // 修正边界检查
            return 0;
        }
        // 注意：这里假设输入是有效的三角形，即 triangle[0].length == 1
        
        if (row == 1) {
            return triangle[0][0];
        }

        List<Integer> pathSums = new ArrayList<>(); // 修正变量名
        dfs(triangle, pathSums, 0, 0, 0);

        return Collections.min(pathSums); // 修正变量名
    }

    public void dfs(int[][] triangle, List<Integer> pathSums, int row, int col, int currentSum) {
        if (row == triangle.length) {
            pathSums.add(currentSum);
            return;
        }

        // 正确的逻辑：只递归调用两个相邻的下层节点
        dfs(triangle, pathSums, row + 1, col, currentSum + triangle[row][col]);
        if (col + 1 < triangle[row].length) { // 确保不越界
             dfs(triangle, pathSums, row + 1, col + 1, currentSum + triangle[row][col]);
        }
        // 注意：在这里显式撤销 currentSum 是多余的，因为它是按值传递的
        // 每次递归调用都会收到当前的 currentSum 值
    }
}

import java.util.List;

class Solution {
    public int minimumTotal(List<List<Integer>> triangle) {
        if (triangle == null || triangle.isEmpty()) {
            return 0;
        }
        int n = triangle.size();
        // 使用一维 DP 数组
        int[] dp = new int[n];
        
        // 初始化 DP 数组的第一行为三角形的顶点值
        dp[0] = triangle.get(0).get(0);

        // 从第二行开始迭代
        for (int i = 1; i < n; i++) {
            // 关键：从后往前更新，防止覆盖尚未使用的 dp[j-1]
            for (int j = i; j >= 0; j--) { 
                int currentValue = triangle.get(i).get(j);
                
                if (j == 0) {
                    // 左边界：只能从上一行的 [j] (也就是 [0]) 来
                    dp[j] = dp[j] + currentValue;
                } else if (j == i) {
                    // 右边界：只能从上一行的 [j-1] (也就是 [i-1]) 来
                    dp[j] = dp[j-1] + currentValue;
                } else {
                    // 普通位置：可以从上一行的 [j-1] 或 [j] 来
                    dp[j] = Math.min(dp[j-1], dp[j]) + currentValue;
                }
            }
        }

        // 在 DP 数组的最后一行（现在就在 dp 数组里）找最小值
        int minPathSum = dp[0];
        for (int j = 1; j < n; j++) {
            minPathSum = Math.min(minPathSum, dp[j]);
        }
        return minPathSum;
    }
}
