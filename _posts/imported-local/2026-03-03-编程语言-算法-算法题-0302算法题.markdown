题目编号： LeetCode 140
题目名称： 单词拆分 II
难度： 🔴 困难
标签： 字符串、动态规划、回溯、记忆化搜索
📝 题目描述
给定一个字符串 s 和一个字符串列表 wordDict 作为字典，将 s 拆分成一个或多个不在字典里的单词，返回所有这些句子。

 考察能力
回溯（Backtracking）思想
动态规划（Dynamic Programming）思想
记忆化搜索（Memoization）
字符串处理
与“单词拆分 I”（只需判断是否可拆分）的区别

import java.util.*;

class Solution {
    // memo 用于记忆化，key 是 start_index, value 是从该索引开始能构成的所有句子列表
    Map<Integer, List<String>> memo;
    Set<String> wordSet;

    public List<String> wordBreak(String s, String[] wordDict) {
        // 初始化成员变量
        this.wordSet = new HashSet<>();
        for (String word : wordDict) {
            this.wordSet.add(word);
        }
        this.memo = new HashMap<>();
        // 开始回溯
        return backtrack(s, 0);
    }

    /**
     * 返回从 s 的 start_index 开始到末尾的所有可能的拆分句子列表
     */
    private List<String> backtrack(String s, int start_index) {
        // 1. 检查备忘录
        if (memo.containsKey(start_index)) {
            return memo.get(start_index);
        }

        // 2. 基本情况：如果到达字符串末尾，返回一个包含空字符串的列表
        // 这是为了在递归的回溯过程中，能够正确地拼接单词。
        if (start_index == s.length()) {
            List<String> res = new ArrayList<>();
            res.add(""); // 一个空字符串，代表此处可以“结束”
            return res;
        }

        List<String> result = new ArrayList<>();

        // 3. 状态转移：枚举所有可能的下一个单词
        for (int end_index = start_index + 1; end_index <= s.length(); end_index++) {
            String word = s.substring(start_index, end_index);
            if (wordSet.contains(word)) {
                // 如果当前单词在字典中，则递归处理剩余部分
                List<String> sub_results = backtrack(s, end_index);
                
                // 将当前单词与子问题的所有结果进行拼接
                for (String sub_sentence : sub_results) {
                    // 注意处理空字符串和加空格的逻辑
                    result.add(word + (sub_sentence.isEmpty() ? "" : " " + sub_sentence));
                }
            }
        }

        // 4. 将结果存入备忘录并返回
        memo.put(start_index, result);
        return result;
    }
}
