=====标准答案======
偷当前节点：那么我就不能偷它的左右孩子节点。当前节点能获得的最大金额是 当前节点值 + 不偷左孩子的最大金额 + 不偷右孩子的最大金额。
不偷当前节点：那么我就可以自由决定是否偷它的左右孩子。当前节点能获得的最大金额是 偷或不偷左孩子的最大金额 + 偷或不偷右孩子的最大金额。
为了实现这个逻辑，我需要一个递归函数，它能返回一个包含两个信息的结构：一个是以当前节点为根的子树，不偷当前节点能获得的最大金额；另一个是偷当前节点能获得的最大金额。这样，父节点可以根据子节点返回的这两个信息来计算自己的最大金额。


/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int rob(TreeNode root) {
        // getRobResult 返回一个数组 [f(node), g(node)]
        int[] result = getRobResult(root);
        // 返回根节点偷与不偷的最大值
        return Math.max(result[0], result[1]);
    }

    /**
     * 递归函数，返回以 node 为根的子树的 [f(node), g(node)] 结果
     * @param node 当前节点
     * @return int[] 数组，[0] 为偷当前节点的最大金额，[1] 为不偷当前节点的最大金额
     */
    private int[] getRobResult(TreeNode node) {
        if (node == null) {
            // 空节点，偷与不偷都为 0
            return new int[]{0, 0};
        }

        // 递归获取左子树的结果
        int[] leftResult = getRobResult(node.left);
        // 递归获取右子树的结果
        int[] rightResult = getRobResult(node.right);

        // f(node) = node.val + g(left) + g(right)
        int robCurrent = node.val + leftResult[1] + rightResult[1];
        // g(node) = max(f(left), g(left)) + max(f(right), g(right))
        int notRobCurrent = Math.max(leftResult[0], leftResult[1]) + Math.max(rightResult[0], rightResult[1]);

        // 返回当前节点的 [f(node), g(node)]
        return new int[]{robCurrent, notRobCurrent};
    }
}
