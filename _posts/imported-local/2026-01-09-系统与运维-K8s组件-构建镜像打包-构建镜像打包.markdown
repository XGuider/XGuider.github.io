构建Dockerfile文件

FROM openjdk:8-jre
ADD target/test-1.0-SNAPSHOT-jar-with-dependencies.jar /application.jar
ENTRYPOINT ["java", "-jar","/application.jar"]

构建镜像：

./build.sh
构建镜像：
APPNAME="hbase-test"
TAG_DAY=`date +"%Y%m%d"`
TAG_NO="001"
TAG=${TAG_DAY}${TAG_NO}
IMAGE_NAME="${APPNAME}:${TAG}"
echo IMAGE_NAME:${IMAGE_NAME}
docker build -t="${IMAGE_NAME}" .
### upload
ARTIFACT_URL="sea.hub:5000"
echo "TAG URL: ${ARTIFACT_URL}/${IMAGE_NAME}"
### tag image
docker tag ${IMAGE_NAME} ${ARTIFACT_URL}/${IMAGE_NAME}
### push image
docker push ${ARTIFACT_URL}/${IMAGE_NAME}

构建deployment：

apiVersion: apps/v1
kind: Deployment
metadata:
  name: hbase-test
  labels:
    app: hbase-test
spec:
  replicas: 1
  selector:
    matchLabels:
      app: hbase-test
  template:
    metadata:
      labels:
        app: hbase-test
    spec:
      containers:
      - name: hbase-test
        image: sea.hub:5000/hbase-test:20250617001
        imagePullPolicy: IfNotPresent
        ports:
        - containerPort: 36066
        volumeMounts:
        - name: hbase-test-volume
          mountPath: /app/config/master/user.keytab
          subPath: user.keytab
        - name: hbase-test-volume
          mountPath: /app/config/master/krb5.conf
          subPath: krb5.conf
      volumes:
      - configMap:
          name: hbase-test--config
          items:
          - path: user.keytab
            key: user.keytab
          - path: krb5.conf
            key: krb5.conf
        name: hbase-test-volume
