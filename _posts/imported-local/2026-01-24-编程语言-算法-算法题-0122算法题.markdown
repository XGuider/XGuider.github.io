给定一个包含非负整数的 m x n 网格 grid，请找出一条从左上角到右下角的路径，使得路径上的数字总和为最小。
说明： 一个机器人每次只能向下或者向右移动一步。

示例 1：
输入： grid = [[1,3,1],[1,5,1],[4,2,1]]
输出： 7
解释： 因为路径 1 → 3 → 1 → 1 → 1 的总和最小。
示例 2：
输入： grid = [[1,2,3],[4,5,6]]
输出： 12




dp[i][j] = Math.min(dp[i][j-1], dp[i-1][j]) + grid[i][j];


# 动态规划
public class Solution {
    public int minPathSum(int[][] grid) { // 1. 修正：返回类型为 int
        if (grid == null || grid.length == 0 || grid[0].length == 0) {
            return 0; // 处理空网格
        }

        int m = grid.length;
        int n = grid[0].length;
        int[][] dp = new int[m][n];

        for (int i = 0; i < m; i++) { // 修正：循环条件 j < n
            for (int j = 0; j < n; j++) { // 2. 修正：循环条件 j < grid[0].length
                if (i == 0 && j == 0) {
                    dp[0][0] = grid[0][0];
                } else if (i == 0) { // 3. 修正：使用 else if，确保 j > 0
                    dp[0][j] = dp[0][j-1] + grid[0][j];
                } else if (j == 0) { // 4. 修正：使用 else if，确保 i > 0
                    dp[i][0] = dp[i-1][0] + grid[i][0];
                } else {
                    dp[i][j] = Math.min(dp[i-1][j], dp[i][j-1]) + grid[i][j]; // 5. 修正：交换 min 的两个参数顺序
                }
            }
        }
        // 6. 修正：将 return 语句移到循环外部
        return dp[m-1][n-1]; 
    }
}
