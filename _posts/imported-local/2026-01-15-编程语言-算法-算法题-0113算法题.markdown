💡 示例
示例 1：

class Solution {
    // 定义四个方向的偏移量，方便后续计算邻居坐标
    private static final int[][] DIRECTIONS = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}}; 

    public boolean exist(char[][] board, String word) {
        if (board == null || board.length == 0 || board[0].length == 0 || word == null || word.length() == 0) {
            return false;
        }

        int rows = board.length;
        int cols = board[0].length;
        // 创建一个布尔矩阵来标记访问过的单元格
        boolean[][] visited = new boolean[rows][cols];

        // 遍历棋盘的每一个单元格作为潜在的起点
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                // 如果从当前单元格出发能找到单词，则返回 true
                if (dfs(board, word, 0, i, j, visited)) {
                    return true;
                }
            }
        }
        // 遍历完所有起点都没找到，返回 false
        return false;
    }

    /**
     * DFS 回溯函数
     * @param board 棋盘
     * @param word 目标单词
     * @param index 当前需要匹配的 word 中的字符索引
     * @param r 当前行
     * @param c 当前列
     * @param visited 访问标记矩阵
     * @return 从 (r, c) 出发是否能找到 word.substring(index)
     */
    private boolean dfs(char[][] board, String word, int index, int r, int c, boolean[][] visited) {
        
        // 1. 边界检查：当前位置是否越界
        if (r < 0 || r >= board.length || c < 0 || c >= board[0].length) {
            return false;
        }
        // 2. 访问检查：当前位置是否已被访问
        if (visited[r][c]) {
            return false;
        }
        // 3. 字符匹配检查：当前位置的字符是否与目标字符匹配
        if (board[r][c] != word.charAt(index)) {
            return false;
        }
        
        // 如果以上三个条件都通过，且已经匹配到了单词的最后一个字符，则成功
        if (index == word.length() - 1) {
            return true;
        }

        // 4. 标记当前位置为已访问
        visited[r][c] = true;

        // 5. 向四个方向进行递归探索
        for (int[] dir : DIRECTIONS) {
            int nextR = r + dir[0];
            int nextC = c + dir[1];
            
            if (dfs(board, word, index + 1, nextR, nextC, visited)) {
                // 如果从邻居找到了完整路径，立即返回 true，无需继续探索其他邻居
                return true;
            }
        }

        // 6. 回溯：如果从当前位置的所有邻居都无法找到完整路径，则撤销访问标记
        visited[r][c] = false;

        // 7. 返回 false，表示从当前位置出发无法找到路径
        return false;
    }
}
