给定一个字符串 s，找到它的第一个不重复的字符，并返回它的索引。如果不存在这样的字符，则返回 -1


class Solution:
    def firstUniqChar(self, s: str) -> int:
        # 第一步：统计频次
        freq = {}
        for char in s:
            freq[char] = freq.get(char, 0) + 1
        
        # 第二步：找第一个频次为1的字符
        for i, char in enumerate(s):
            if freq[char] == 1:
                return i
        
        return -1
