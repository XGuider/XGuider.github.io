压缩：

gs -sDEVICE=pdfwrite -dCompatibilityLevel=1.4 \
  -dPDFSETTINGS=/screen -dNOPAUSE -dQUIET -dBATCH \
  -sOutputFile=扫描件_劳动合同书2.pdf 扫描件_劳动合同书.pdf
